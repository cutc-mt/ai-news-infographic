import fal_client

def generate_image_with_seedream():
    result = fal_client.run(
        "fal-ai/seedream-4.0",
        arguments={
            "prompt": "カレーの作り方を手書き風で作成して",
            "resolution": "1024x1024",
            "style": "handwritten_recipe"
        }
    )
    return result
